## 代码

前序迭代：
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:  // 前序迭代
    vector<int> preorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> result;

        if(root == NULL) return NULL;
        st.push(root); // 推入栈

        while(!st.empty()){
            TreeNode* node = st.top();
            st.pop();
            result.pusn_back(node->val); // 先把中存入result

            if(node->right){ // 先进右孩子，因为栈先进后出
                st.push(node->right);
            }
            if(node->left){ // 再进左孩子
                st.push(node->left);
            }
        }

        return result;
    }
};
```

中序迭代：

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public: // 中序迭代
    vector<int> inorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> result;

        if(root == NULL) return result;
        TreeNode* cur = root; // 创建指针来操作遍历

        while(cur != NULL || !st.empty()){ // 指针不为空，栈也不为空
            if(cur != NULL){ // 指针不为空，就存入栈，然后向左一直探索到底
                st.push(cur);
                cur = cur->left;
            }
            else{ // 遇到指针为空，则说明到底
                cur = st.top(); // 释放栈顶元素
                st.pop();
                result.push_back(cur->val); // 并加入结果数组
                cur = cur->right; // 向右探索
            }
        }

        return result;
    }
};
```


后序迭代：

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public: // 后序迭代 左右中————>翻转  中右左，得到此目标即可
    vector<int> postorderTraversal(TreeNode* root) {
        stack<TreeNode*> st;
        vector<int> result;

        if(root == NULL) return result;
        st.push(root);

        while(!st.empty()){
            TreeNode* node = st.top();
            st.pop();
            result.push_back(node->val);

            if(node->left) st.push(node->left); // 先进左，左后出
            if(node->right) st.push(node->right); // 右先出，因为目标是中右左
        }

        // 已经得到中右左，翻转一下就是左右中
        reverse(result.begin(), result.end());

        return result;
    }
};
```
